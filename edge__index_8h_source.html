525 template <typename SrcT, typename DestT, typename StorageT>
526 inline void EdgeIndex<SrcT, DestT, StorageT>::InsertEdgeList(
527  const SrcT &src, DestT *destinations, size_t numDest, bool overwrite) {
528  size_t targetId = shad::hash<SrcT>{}(src) % rt::numLocalities();
529  rt::Locality targetLocality(targetId);
530  if (targetLocality == rt::thisLocality()) {
531  localIndex_.InsertEdgeList(src, destinations, numDest, overwrite);
532  } else {
533  int toInsert = numDest;
534  auto insertLambda = [](const EdgeListChunk &args) {
535  auto ptr = EdgeIndex<SrcT, DestT, StorageT>::GetPtr(args.oid);
536  ptr->localIndex_.Insert(args.src, args.chunk);
537  };
538  size_t locSize = StorageT::kEdgeListChunkSize_;
539  size_t chunkSize = std::min<size_t>(locSize, toInsert);
540  LocalEdgeListChunk lchunk(numDest, overwrite, destinations);
541  EdgeListChunk args(oid_, src, lchunk);
542  rt::executeAt(targetLocality, insertLambda, args);
543  toInsert -= chunkSize;
544  while (toInsert > 0) {
545  destinations += chunkSize;
546  chunkSize = std::min<size_t>(locSize, toInsert);
547  LocalEdgeListChunk lchunk(chunkSize, false, destinations);
548  EdgeListChunk args(oid_, src, lchunk);
549  rt::executeAt(targetLocality, insertLambda, args);
550  toInsert -= chunkSize;
551  }
552  }
553 }

564  } else {

566  auto ptr = EdgeIndex<SrcT, DestT, StorageT>::GetPtr(args.oid);
567  ptr->localIndex_.Insert(args.src, args.chunk);
568  };
569  auto insertLambda = [](rt::Handle &handle, const EdgeListChunk &args) {
570  auto ptr = EdgeIndex<SrcT, DestT, StorageT>::GetPtr(args.oid);
571  ptr->localIndex_.AsyncInsert(handle, args.src, args.chunk);
572  };
573

576  size_t chunkSize = std::min<size_t>(locSize, toInsert);
577  LocalEdgeListChunk lchunk(numDest, overwrite, destinations);
578  EdgeListChunk args(oid_, src, lchunk);
579  if (numDest <= locSize) {
580  rt::asyncExecuteAt(handle, targetLocality, insertLambda, args);
581  return;

583  rt::executeAt(targetLocality, syncInsertLambda, args);
584 
585  toInsert -= chunkSize;
586  while (toInsert > 0) {
587  destinations += chunkSize;
588  chunkSize = std::min<size_t>(locSize, toInsert);
589  lchunk = LocalEdgeListChunk(chunkSize, false, destinations);
590  args = EdgeListChunk(oid_, src, lchunk);
591  rt::asyncExecuteAt(handle, targetLocality, insertLambda, args);
592  toInsert -= chunkSize;
593  }
594  }
