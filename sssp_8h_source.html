86 template <typename GraphT, typename VertexT>
87 size_t __sssp_length( // GraphT::SharedPtr gPtr,
88  typename GraphT::ObjectID gid, size_t num_vertices,
89  typename shad::Set<VertexT>::SharedPtr to_visit_0,
90  typename shad::Set<VertexT>::SharedPtr to_visit_1,
91  shad::Array<bool>::SharedPtr visitedPtr,
92  shad::Array<bool>::SharedPtr foundPtr, VertexT src, VertexT dest) {
93  if (src == dest) return 0;
94  size_t level = 0;
95  shad::Set<size_t>::ShadSetPtr qPtr, nextqPtr;
96  qPtr = to_visit_0;
97  nextqPtr = to_visit_1;
98 
99  qPtr->Insert(src);
100  bool v = true;
101  visitedPtr->InsertAt(src, v);
102  auto visitedID = visitedPtr->GetGlobalID();
103  auto foundID = foundPtr->GetGlobalID();
104  shad::rt::Handle handle;
105  while (qPtr->Size()) {
106  auto nextID = nextqPtr->GetGlobalID();
107  qPtr->AsyncForEachElement(handle, __sssp_iteration<GraphT, VertexT>, gid,
108  nextID, visitedID, foundID, dest);
109  shad::rt::waitForCompletion(handle);
110  // check solution
111  ++level;
112  if (foundPtr->At(0)) return level;
113  // prepare for next round
114  qPtr->Reset(num_vertices / 2);
115  qPtr.swap(nextqPtr);
116  }
117  return std::numeric_limits<size_t>::infinity();
118 }
