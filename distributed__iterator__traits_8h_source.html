116  // split a range into at most n sub-ranges
117  static std::vector<partition_range> partitions(T *begin, T *end, size_t n) {
118  auto range_len = std::distance(begin, end);
119  auto n_parts = std::min(n, static_cast<size_t>(range_len));
120  std::vector<partition_range> res;
121  if (n_parts) {
122  auto block_size = (range_len + n_parts - 1) / n_parts;
123  for (size_t block_id = 0; block_id < n_parts; ++block_id) {
124  auto block_begin = begin;
125  std::advance(block_begin, block_id * block_size);
126  auto block_end = block_begin;
127  if (std::distance(block_begin, end) < block_size) {
128  if (block_begin != end) {
129  res.push_back(partition_range{block_begin, end});
130  }
131  break;
132  } else {
133  std::advance(block_end, block_size);
134  assert(block_begin != block_end);
135  res.push_back(partition_range{block_begin, block_end});
136  }
137  }
138  assert(res.back().end() == end);
139  }
140  return res;
141  }
142 };
