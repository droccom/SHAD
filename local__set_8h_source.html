359 bool LocalSet<T, ELEM_COMPARE>::Find(const T& element) {
360  size_t bucketIdx = shad::hash<T>{}(element) % numBuckets_;
361  Bucket* bucket = &(buckets_array_[bucketIdx]);
362 
363  while (bucket != nullptr) {
364  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
365  Entry* entry = &bucket->getEntry(i);
366  // Stop at the first empty entry.
367  if (entry->state == EMPTY) return false;
368  // Yield on pending entries.
369  while (entry->state == PENDING_INSERT) rt::impl::yield();
370  // Entry is USED.
371  if (ElemComp_(&entry->element, &element) == 0) {
372  return true;
373  }
374  }
375  bucket = bucket->next.get();
376  }
377  return false;

400 template <typename T, typename ELEM_COMPARE>
401 void LocalSet<T, ELEM_COMPARE>::Erase(const T& element) {
402  size_t bucketIdx = shad::hash<T>{}(element) % numBuckets_;
403  Bucket* bucket = &(buckets_array_[bucketIdx]);
404  Entry* prevEntry = nullptr;
405  Entry* toDelete = nullptr;
406  Entry* lastEntry = nullptr;
407  for (;;) {
408  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
409  Entry* entry = &bucket->getEntry(i);
410  // 1. Key not found, returning
411  if (entry->state == EMPTY) {
412  if (toDelete != nullptr)
413  throw std::logic_error(
414  "A problem occured with"
415  "the set erase operation");
416  return;
417  }
418  while (entry->state == PENDING_INSERT) {
419  rt::impl::yield();
420  }
421  if (ElemComp_(&entry->element, &element) == 0) {
422  // 2. Key found, try to acquire a lock on it
423  if (!__sync_bool_compare_and_swap(&entry->state, USED,
424  PENDING_INSERT)) {
425  Erase(element);
426  return;
427  }
428  // 3. The entry to remove has been found,
429  // and its status set to PENDING_INSERT
430  toDelete = entry;
431  prevEntry = entry;
432  --size_;
433 
434  // now look for the last two entries.
435  size_t j = i + 1;
436  for (;;) {
437  for (; j < bucket->BucketSize(); ++j) {
438  lastEntry = &bucket->getEntry(j);
439  if (__sync_bool_compare_and_swap(&lastEntry->state, EMPTY,
440  PENDING_INSERT)) {
441  // 3. Last entry found (EMPTY->PENDING)
442  if (prevEntry == toDelete) { // just set it to EMPTY and return;
443  lastEntry->state = EMPTY;
444  toDelete->state = EMPTY;
445  return;
446  }
447  // STATUS:
448  // toDelete found and status is PENDING_INSERT
449  // lastEntry found and status is PENDING_INSERT
450  // need to find prevEntry
451 
452  if (!__sync_bool_compare_and_swap(&prevEntry->state, USED,
453  PENDING_INSERT)) {
454  rt::impl::yield();
455  lastEntry->state = EMPTY;
456  toDelete->state = USED;
457  ++size_;
458  Erase(element);
459  return;
460  }
461  // now prevEntry is locked
462  // 4. free the last entry
463  lastEntry->state = EMPTY;
464  // move prevEntry into toDelete
465  toDelete->element = std::move(prevEntry->element);
466  toDelete->state = USED;
467  // free prevEntry
468  prevEntry->state = EMPTY;
469  return;
470  } else {
471  if (lastEntry->state == PENDING_INSERT) {
472  toDelete->state = USED;
473  ++size_;
474  Erase(element);
475  return;
476  }
477  }
478  prevEntry = lastEntry;
479  }
480  j = 0;
481  if (bucket->next != nullptr) {
482  bucket = bucket->next.get();
483  } else {
484  // STATUS last entry is not empty and has not been locked
485  if (lastEntry == nullptr) {
486  // toDelete has not been found or
487  // it is the last entry at the end of the last bucket
488  if (toDelete != nullptr) toDelete->state = EMPTY;
489  return;
490  }
491  if (!__sync_bool_compare_and_swap(&lastEntry->state, USED,
492  PENDING_INSERT)) {
493  toDelete->state = USED;
494  ++size_;
495  Erase(element);
496  return;
497  }
498  if (lastEntry == prevEntry) {
499  if (toDelete == prevEntry) {
500  // No move is necessary, just set to EMPTY
501  lastEntry->state = EMPTY;
502  toDelete->state = EMPTY;
503  return;
504  } else {
505  toDelete->element = std::move(lastEntry->element);
506  toDelete->state = USED;
507  lastEntry->state = EMPTY;
508  return;
509  }
510  } else {
511  // FIXME double check if this state is reachable
512  if (toDelete == prevEntry) {
513  toDelete->element = std::move(lastEntry->element);
514  toDelete->state = USED;
515  lastEntry->state = EMPTY;
516  return;
517  } else {
518  // Need to lock prev entry as well
519  while (!__sync_bool_compare_and_swap(&prevEntry->state, USED,
520  PENDING_INSERT)) {
521  rt::impl::yield();
522  }
523  lastEntry->state = EMPTY;
524  toDelete->element = std::move(prevEntry->element);
525  toDelete->state = USED;
526  prevEntry->state = EMPTY;
527  }
528  }
529  return;
530  }
531  }
532  }
533  prevEntry = entry;
534  }
535  if (bucket->next != nullptr) {
536  bucket = bucket->next.get();
537  } else {
538  return;
539  }
540  }

554 template <typename T, typename ELEM_COMPARE>
555 void LocalSet<T, ELEM_COMPARE>::Insert(const T& element) {
556  size_t bucketIdx = shad::hash<T>{}(element) % numBuckets_;
557  Bucket* bucket = &(buckets_array_[bucketIdx]);
558 
559  // Forever or until we find an insertion point.
560  for (;;) {
561  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
562  Entry* entry = &bucket->getEntry(i);
563 
564  if (__sync_bool_compare_and_swap(&entry->state, EMPTY, PENDING_INSERT)) {
565  entry->element = std::move(element);
566  ++size_;
567  entry->state = USED;
568  return;
569  } else {
570  while (entry->state == PENDING_INSERT) rt::impl::yield();
571  if (ElemComp_(&entry->element, &element) == 0) {
572  return;
573  }
574  }
575  }
576 
577  if (bucket->next == nullptr) {
578  // We need to allocate a new buffer
579  if (__sync_bool_compare_and_swap(&bucket->isNextAllocated, false, true)) {
580  // Allocate the bucket
581  std::shared_ptr<Bucket> newBucket(
582  new Bucket(constants::kSetDefaultNumEntriesPerBucket));
583  bucket->next.swap(newBucket);
584  } else {
585  // Wait for the allocation to happen
586  while (bucket->next == nullptr) rt::impl::yield();
587  }
588  }
589 
590  bucket = bucket->next.get();
591  }
592 }
