669 template <typename KTYPE, typename VTYPE, typename KEY_COMPARE,
670  typename INSERTER>
671 void LocalHashmap<KTYPE, VTYPE, KEY_COMPARE, INSERTER>::Erase(
672  const KTYPE &key) {
673  size_t bucketIdx = shad::hash<KTYPE>{}(key) % numBuckets_;
674  Bucket *bucket = &(buckets_array_[bucketIdx]);
675  Entry *prevEntry = nullptr;
676  Entry *toDelete = nullptr;
677  Entry *lastEntry = nullptr;
678  auto printEntryState = [](size_t num, Entry *todel, Entry *last,
679  Entry *prev) {
680  size_t tds = todel->state, ls = 42, ps = 42;
681  if (last != nullptr) ls = last->state;
682  if (prev != nullptr) ps = prev->state;
683  printf("loop %lu, todel-s: %lu, last-s: %lu, prev-s: %lu\n", num, tds, ls,
684  ps);
685  };
686  for (;;) {
687  for (size_t i = 0; i < bucket->BucketSize(); ++i) {
688  Entry *entry = &bucket->getEntry(i);
689 
690  // 1. Key not found, returning
691  if (entry->state == EMPTY) {
692  if (toDelete != nullptr)
693  throw std::logic_error(
694  "A problem occured with"
695  "the map erase operation");
696  return;
697  }
698  while (entry->state == PENDING_INSERT) {
699  rt::impl::yield();
700  }
701 
702  if (KeyComp_(&entry->key, &key) == 0) {
703  // 2. Key found, try to acquire a lock on it
704  if (!__sync_bool_compare_and_swap(&entry->state, USED,
705  PENDING_INSERT)) {
706  // entry has already been deleted by another operation
707  Erase(key);
708  return;
709  }
710  // 3. The entry to remove has been found,
711  // and its status set to PENDING_INSERT
712  toDelete = entry;
713  prevEntry = entry;
714  size_--;
715 
716  // now look for the last two entries.
717  size_t j = i + 1;
718  for (;;) {
719  size_t numBuck = 0;
720  for (; j < bucket->BucketSize(); ++j) {
721  lastEntry = &bucket->getEntry(j);
722  if (__sync_bool_compare_and_swap(&lastEntry->state, EMPTY,
723  PENDING_INSERT)) {
724  // 3. Last entry found (EMPTY->PENDING)
725  if (prevEntry == toDelete) { // just set it to EMPTY and return;
726  lastEntry->state = EMPTY;
727  toDelete->state = EMPTY;
728  return;
729  }
730  // STATUS:
731  // toDelete found and status is PENDING_INSERT
732  // lastEntry found and status is PENDING_INSERT
733  // need to find prevEntry
734 
735  if (!__sync_bool_compare_and_swap(&prevEntry->state, USED,
736  PENDING_INSERT)) {
737  printEntryState(2, toDelete, lastEntry, prevEntry);
738  rt::impl::yield();
739  lastEntry->state = EMPTY;
740  toDelete->state = USED;
741  size_++;
742  Erase(key);
743  return;
744  }
745  // now prevEntry is locked
746  // 4. free the last entry
747  lastEntry->state = EMPTY;
748  // move prevEntry into toDelete
749  toDelete->key = std::move(prevEntry->key);
750  toDelete->value = std::move(prevEntry->value);
751  toDelete->state = USED;
752  // free prevEntry
753  prevEntry->state = EMPTY;
754  return;
755  } else {
756  if (lastEntry->state == PENDING_INSERT) {
757  toDelete->state = USED;
758  size_++;
759  Erase(key);
760  return;
761  }
762  }
763  prevEntry = lastEntry;
764  }
765  j = 0;
766  numBuck++;
767  if (bucket->next != nullptr) {
768  bucket = bucket->next.get();
769  } else {
770  // STATUS last entry is not empty and has not been locked
771  if (lastEntry == nullptr) {
772  // toDelete has not been found or
773  // it is the last entry at the end of the last bucket
774  if (toDelete != nullptr) toDelete->state = EMPTY;
775  return;
776  }
777  if (!__sync_bool_compare_and_swap(&lastEntry->state, USED,
778  PENDING_INSERT)) {
779  toDelete->state = USED;
780  size_++;
781  Erase(key);
782  return;
783  }
784  if (lastEntry == prevEntry) {
785  if (toDelete == prevEntry) {
786  // No move is necessary, just set to EMPTY
787  lastEntry->state = EMPTY;
